    const performanceMetrics = await this.getPerformanceMetrics(
      startDate,
      endDate
    );
    const costBreakdown = await this.getCostBreakdown(startDate, endDate);
    const transcriptStats = await this.getTranscriptStats(startDate, endDate);
    const audioStats = await this.getAudioStats(startDate, endDate);
 
    return {
      id: crypto.randomUUID(),
      type,
      start_date: startDate.toISOString(),
      end_date: endDate.toISOString(),
      usage_stats: usageStats,
      performance_metrics: performanceMetrics,
      cost_breakdown: costBreakdown,

  async exportReport(report: Report, format: ExportFormat): Promise<Blob> {
    if (format === 'json') {
      return new Blob([JSON.stringify(report, null, 2)], {
        type: 'application/json',
      });
    }
 
    if (format === 'csv') {
      const csv = this.reportToCSV(report);
      return new Blob([csv], { type: 'text/csv' });
    }
 
    if (format === 'pdf') {
      const pdfBlob = await this.reportToPDF(report);
      return pdfBlob;
    }
 
    throw new Error(`Unsupported export format: ${format}`);
  }
 
  private reportToCSV(report: Report): string {
    const lines: string[] = [];
 
    // Header
    lines.push(`Report Type,${report.type}`);
    lines.push(`Start Date,${report.start_date}`);
    lines.push(`End Date,${report.end_date}`);
    lines.push(`Generated At,${report.generated_at}`);
    lines.push('');
 
    // Usage Stats
    lines.push('Usage Statistics');
    lines.push(
      'Date,Transcripts,Audio Uploads,Exports,AI Features,Minutes,Words'
    );
    for (const stat of report.usage_stats) {
      lines.push(
        `${stat.date},${stat.transcripts_count},${stat.audio_uploads_count},${stat.exports_count},${stat.ai_features_count},${stat.total_minutes},${stat.total_words}`
      );
    }
    lines.push('');
 
    // Cost Breakdown
    lines.push('Cost Breakdown');
    lines.push(
      'Date,API Calls,API Cost,Storage GB,Storage Cost,AI Features Cost,Total Cost'
    );
    for (const cost of report.cost_breakdown) {
      lines.push(
        `${cost.date},${cost.api_calls},${cost.api_cost},${cost.storage_gb},${cost.storage_cost},${cost.ai_features_cost},${cost.total_cost}`
      );
    }
 
    return lines.join('\n');
  }
 
  private async reportToPDF(report: Report): Promise<Blob> {
    const doc = new jsPDF();
    let yPos = 20;
    const pageWidth = doc.internal.pageSize.getWidth();
    const margin = 20;
    const lineHeight = 7;
 
    // Helper function to add text with auto page break
    const addText = (
      text: string,
      x: number,
      fontSize: number = 10,
      isBold: boolean = false
    ) => {
      if (yPos > 270) {
        doc.addPage();
        yPos = 20;
      }
      doc.setFontSize(fontSize);
      if (isBold) {
        doc.setFont('helvetica', 'bold');
      } else {
        doc.setFont('helvetica', 'normal');
      }
      doc.text(text, x, yPos);
      yPos += lineHeight;
    };
 
    // Title
    addText('Analytics Report', margin, 18, true);
    yPos += 5;
 
    // Report Info
    addText(`Report Type: ${report.type}`, margin, 12);
    addText(
      `Period: ${new Date(report.start_date).toLocaleDateString()} - ${new Date(report.end_date).toLocaleDateString()}`,
      margin,
      10
    );
    addText(
      `Generated: ${new Date(report.generated_at).toLocaleString()}`,
      margin,
      10
    );
    yPos += 10;
 
    // Usage Statistics Section
    addText('Usage Statistics', margin, 14, true);
    yPos += 3;
 
    if (report.usage_stats.length > 0) {
      // Calculate totals
      const totals = report.usage_stats.reduce(
        (acc, stat) => ({
          transcripts: acc.transcripts + stat.transcripts_count,
          uploads: acc.uploads + stat.audio_uploads_count,
          exports: acc.exports + stat.exports_count,
          aiFeatures: acc.aiFeatures + stat.ai_features_count,
          minutes: acc.minutes + stat.total_minutes,
          words: acc.words + stat.total_words,
        }),
        {
          transcripts: 0,
          uploads: 0,
          exports: 0,
          aiFeatures: 0,
          minutes: 0,
          words: 0,
        }
      );
 
      addText(`Total Transcripts: ${totals.transcripts}`, margin + 5, 10);
      addText(`Total Audio Uploads: ${totals.uploads}`, margin + 5, 10);
      addText(`Total Exports: ${totals.exports}`, margin + 5, 10);
      addText(`Total AI Features Used: ${totals.aiFeatures}`, margin + 5, 10);
      addText(`Total Minutes: ${totals.minutes.toFixed(1)}`, margin + 5, 10);
      addText(`Total Words: ${totals.words.toLocaleString()}`, margin + 5, 10);
    }
    yPos += 10;
 
    // Performance Metrics Section
    addText('Performance Metrics', margin, 14, true);
    yPos += 3;
 
    if (report.performance_metrics.length > 0) {
      const avgMetrics = report.performance_metrics.reduce(
        (acc, metric) => ({
          transcriptionTime: acc.transcriptionTime + metric.avg_latency,
          processingTime: acc.processingTime + metric.avg_latency,
          accuracy: acc.accuracy + metric.avg_accuracy,
          count: acc.count + 1,
        }),
        { transcriptionTime: 0, processingTime: 0, accuracy: 0, count: 0 }
      );
 
      addText(
        `Avg Transcription Time: ${(avgMetrics.transcriptionTime / avgMetrics.count).toFixed(2)}s`,
        margin + 5,
        10
      );
      addText(
        `Avg Processing Time: ${(avgMetrics.processingTime / avgMetrics.count).toFixed(2)}s`,
        margin + 5,
        10
      );
      addText(
        `Avg Accuracy: ${((avgMetrics.accuracy / avgMetrics.count) * 100).toFixed(1)}%`,
        margin + 5,
        10
      );
    }
    yPos += 10;
 
    // Cost Breakdown Section
    addText('Cost Breakdown', margin, 14, true);
    yPos += 3;
 
    if (report.cost_breakdown.length > 0) {
      const totalCosts = report.cost_breakdown.reduce(
        (acc, cost) => ({
          apiCalls: acc.apiCalls + cost.api_calls,
          apiCost: acc.apiCost + cost.api_cost,
          storageGB: acc.storageGB + cost.storage_gb,
          storageCost: acc.storageCost + cost.storage_cost,
          aiCost: acc.aiCost + cost.ai_features_cost,
          total: acc.total + cost.total_cost,
        }),
        {
          apiCalls: 0,
          apiCost: 0,
          storageGB: 0,
          storageCost: 0,
          aiCost: 0,
          total: 0,
        }
      );
 
      addText(
        `Total API Calls: ${totalCosts.apiCalls.toLocaleString()}`,
        margin + 5,
        10
      );
      addText(`API Cost: $${totalCosts.apiCost.toFixed(2)}`, margin + 5, 10);
      addText(
        `Storage: ${totalCosts.storageGB.toFixed(2)} GB ($${totalCosts.storageCost.toFixed(2)})`,
        margin + 5,
        10
      );
      addText(
        `AI Features Cost: $${totalCosts.aiCost.toFixed(2)}`,
        margin + 5,
        10
      );
      addText(
        `Total Cost: $${totalCosts.total.toFixed(2)}`,
        margin + 5,
        10,
        true
      );
    }
    yPos += 10;
 
    // Transcript Statistics
    if (report.transcript_stats) {
      addText('Transcript Statistics', margin, 14, true);
      yPos += 3;
      addText(
        `Total Transcripts: ${report.transcript_stats.total}`,
        margin + 5,
        10
      );
      addText(
        `Avg Duration: ${report.transcript_stats.avg_duration.toFixed(1)} min`,
        margin + 5,
        10
      );
      addText(
        `Avg Word Count: ${Math.round(report.transcript_stats.avg_word_count)}`,
        margin + 5,
        10
      );
      addText(
        `Avg Confidence: ${(report.transcript_stats.avg_confidence * 100).toFixed(1)}%`,
        margin + 5,
        10
      );
      yPos += 10;
    }
 
    // Audio Statistics
    if (report.audio_stats) {
      addText('Audio Statistics', margin, 14, true);
      yPos += 3;
      addText(`Total Files: ${report.audio_stats.total}`, margin + 5, 10);
      addText(
        `Total Size: ${(report.audio_stats.total_size_mb / 1024).toFixed(2)} GB`,
        margin + 5,
        10
      );
      addText(
        `Avg File Size: ${report.audio_stats.avg_size_mb.toFixed(2)} MB`,
        margin + 5,
        10
      );
 
      if (report.audio_stats.by_format) {
        yPos += 5;
        addText('Format Breakdown:', margin + 5, 10, true);
        Object.entries(report.audio_stats.by_format).forEach(
          ([format, count]) => {
            addText(`  ${format.toUpperCase()}: ${count}`, margin + 10, 9);
          }
        );
      }
    }
 
    // Footer
    const pageCount = doc.getNumberOfPages();
    for (let i = 1; i <= pageCount; i++) {
      doc.setPage(i);
      doc.setFontSize(8);
      doc.setFont('helvetica', 'normal');
      doc.text(
        `Page ${i} of ${pageCount}`,
        pageWidth / 2,
        doc.internal.pageSize.getHeight() - 10,
        { align: 'center' }
      );
    }
 
    return doc.output('blob');
  }

  async getScheduledReports(): Promise<ScheduledReport[]> {
    const user = await this.supabase.getCurrentUser();
    if (!user) throw new Error('User not authenticated');
 
    const { data, error } = await this.supabase
      .getClient()
      .from('scheduled_reports')
      .select('*')
      .eq('user_id', user.id)
      .order('created_at', { ascending: false });
 
    if (error) throw error;
    return data || [];
  }

    config: Omit<
      ScheduledReport,
      'id' | 'user_id' | 'created_at' | 'updated_at'
    >
  ): Promise<ScheduledReport> {
    const user = await this.supabase.getCurrentUser();

      .getClient()
      .from('scheduled_reports')
      .insert({
        user_id: user.id,
        ...config,
      })
      .select()
      .single();

    return data;
  }
